KAT-I-0005 — Kanban Board Plugin Interface.

## Goal
End-to-end smoke test that proves the four pieces (port, internal backend, renderer, CLI) compose correctly against a realistic fixture workspace and that the external-stub also renders. This is the initiative's exit-criteria evidence.

## Architecture & Contracts

The test does NOT spin up the MCP server. It:
1. Builds a fake `InternalBoardDeps` by reading fixture frontmatter directly via `gray-matter` and producing `DocumentSummary[]`.
2. Calls `run_board_command({ backend: "internal", workspace_root: fixturePath }, { internal_deps, stdout })`.
3. Replaces the dynamic `generated_at` line with a fixed marker before comparing to the golden file.
4. Repeats for `backend: "external-stub"`.

```ts
// tests/board/smoke.test.ts (sketch)
import { describe, it, expect } from "vitest";
import * as fs from "node:fs";
import * as path from "node:path";
import matter from "gray-matter";
import { run_board_command } from "../../src/cli/board.js";
import type { InternalBoardDeps, DocumentSummary } from "../../src/board/internal-deps.js";

function loadFixtureDeps(root: string): InternalBoardDeps {
  const files = fs.readdirSync(root).filter((f) => f.endsWith(".md"));
  const summaries: DocumentSummary[] = files.map((f) => {
    const fm = matter(fs.readFileSync(path.join(root, f), "utf8")).data as any;
    return {
      short_code: fm.short_code,
      level: fm.level,
      title: fm.title,
      phase: fm.phase ?? fm.tags?.find((t: string) => t.startsWith("#phase/"))?.slice(7) ?? "todo",
      parent: fm.parent,
      archived: !!fm.archived,
      updated_at: fm.updated_at,
    };
  });
  return {
    list_documents: async (opts) => summaries.filter((s) =>
      (!opts.level || s.level === opts.level) &&
      (!opts.parent || s.parent === opts.parent) &&
      (opts.include_archived || !s.archived)
    ),
    transition_phase: async () => { throw new Error("not used in smoke"); },
  };
}

const STABLE = "GENERATED_AT_REPLACED";
function stabilize(md: string): string {
  return md.replace(/generated_at: [^_]+_/, `generated_at: ${STABLE}_`);
}

describe("board smoke", () => {
  const root = path.join(__dirname, "fixtures/workspace");
  it("internal backend matches golden", async () => {
    const out: string[] = [];
    const code = await run_board_command(
      { backend: "internal", workspace_root: root, hide_empty: false },
      { internal_deps: loadFixtureDeps(root), stdout: (s) => out.push(s) },
    );
    expect(code).toBe(0);
    const expected = fs.readFileSync(path.join(__dirname, "fixtures/expected-internal.md"), "utf8");
    expect(stabilize(out.join(""))).toBe(stabilize(expected));
  });

  it("external-stub backend matches golden", async () => {
    const out: string[] = [];
    const code = await run_board_command(
      { backend: "external-stub", workspace_root: root },
      { internal_deps: loadFixtureDeps(root), stdout: (s) => out.push(s) },
    );
    expect(code).toBe(0);
    const expected = fs.readFileSync(path.join(__dirname, "fixtures/expected-stub.md"), "utf8");
    expect(stabilize(out.join(""))).toBe(stabilize(expected));
  });
});
```
